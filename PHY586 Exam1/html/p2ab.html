b)
f1=T1*sin(theta_1-T2*sin(theta_2)-m1g
f2=T1*cos(theta1)-T2*cos(theta2)=0
f3=T2*sin(theta2)+T3*sin(theta3)-m2g=0
f4=T2*cos(theta2)-T3*cos(theta3)=0
f5=L1*cos(theta)+L2*cos(theta2)+L3*cos(theta3)-L=0
f6=L1*sin(theta)+L2*sin(theta2)-L3*sin(theta3)=0
fj=sin^2(thetai)+cos^2(thetai)-1=0, for j=7,8,9 and i=1,2,3.
